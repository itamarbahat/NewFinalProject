# יצירת הסקריפט

צור קובץ חדש:

```
src/tools/plot_experiments.py
```

והדבק בו את הקוד הבא.

```python
import os
import pandas as pd
import matplotlib.pyplot as plt

RUNS_DIR = "runs"
OUTPUT_DIR = "artifacts/figures"

os.makedirs(OUTPUT_DIR, exist_ok=True)

summary = []

for root, dirs, files in os.walk(RUNS_DIR):
    if "metrics.csv" in files:
        run_name = os.path.basename(root)
        path = os.path.join(root, "metrics.csv")

        df = pd.read_csv(path)

        if "val_acc" not in df.columns:
            continue

        # ---- Accuracy graph ----
        plt.figure()
        plt.plot(df["epoch"], df["train_acc"], label="train")
        plt.plot(df["epoch"], df["val_acc"], label="val")
        plt.title(f"Accuracy - {run_name}")
        plt.xlabel("Epoch")
        plt.ylabel("Accuracy")
        plt.legend()
        plt.grid()

        plt.savefig(f"{OUTPUT_DIR}/{run_name}_accuracy.png")
        plt.close()

        # ---- Loss graph ----
        if "train_loss" in df.columns:
            plt.figure()
            plt.plot(df["epoch"], df["train_loss"], label="train")
            if "val_loss" in df.columns:
                plt.plot(df["epoch"], df["val_loss"], label="val")

            plt.title(f"Loss - {run_name}")
            plt.xlabel("Epoch")
            plt.ylabel("Loss")
            plt.legend()
            plt.grid()

            plt.savefig(f"{OUTPUT_DIR}/{run_name}_loss.png")
            plt.close()

        best_acc = df["val_acc"].max()

        summary.append({
            "run": run_name,
            "best_val_acc": best_acc
        })

# ---- Summary table ----
summary_df = pd.DataFrame(summary)
summary_df = summary_df.sort_values("best_val_acc", ascending=False)

summary_df.to_csv("artifacts/tables/run_summary_auto.csv", index=False)

print("Graphs saved to artifacts/figures/")
print("Summary table saved to artifacts/tables/run_summary_auto.csv")
```
